# Temporary workaround for the vanilla Platonic Revival multiplayer desync.

# Scope-free: runs from country pulses and the no-scope weather_monthly_pulse.
ht_temp_enforce_hellenism_desync_fix = {
	if = {
		limit = { any_country = { religion = religion:hellenism_religion } }
		ht_temp_convert_hellenism_to_orthodox = yes
	}
}

ht_temp_convert_hellenism_to_orthodox = {
	every_country = {
		if = {
			limit = { religion = religion:hellenism_religion }
			change_religion = religion:orthodox
		}
		every_character = {
			limit = { religion = religion:hellenism_religion }
			change_character_religion = religion:orthodox
		}
		every_owned_location = {
			every_pop = {
				limit = { religion = religion:hellenism_religion }
				change_pop_religion = religion:orthodox
			}
		}
	}
}
