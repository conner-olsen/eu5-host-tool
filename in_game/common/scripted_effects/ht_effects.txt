# Single-target: recipient = country whose regency ends (random local adult takes the throne)
ht_end_regency_effect = {
	scope:recipient = {
		if = {
			limit = {
				any_character = { can_become_a_ruler = yes }
			}
			random_character = {
				limit = { can_become_a_ruler = yes }
				save_scope_as = ht_new_ruler
			}
		}
		else = {
			create_character = {
				age = 35
				religion = scope:recipient.religion
				culture = scope:recipient.culture
				save_scope_as = ht_new_ruler
			}
		}
		set_new_ruler = scope:ht_new_ruler
	}
	scope:recipient = { save_scope_as = cmf_log_arg2 }
	scope:actor = { cmf_log_with_scope_arg = { action = ht_log_ended_regency } }
}

# Single-target: recipient = country whose pops convert to its primary religion
# Per-pop fraction = (CMM% / 100) / non_primary_share, clamped to [0, 1], so the total converted equals CMM% of TOTAL pops.
ht_convert_pops_effect = {
	scope:recipient = {
		every_pop = {
			limit = {
				NOT = { religion = scope:recipient.religion }
			}
			split_pop = {
				fraction = {
					value = "global_variable_map(cmm|flag:ht__religion_conversion_percentage)"
					divide = 100
					divide = {
						value = 1
						subtract = "scope:recipient.religion_percentage_in_country(scope:recipient.religion)"
						min = 0.001
					}
					min = 0
					max = 1
				}
				religion = scope:recipient.religion
			}
		}
	}
	scope:recipient = { save_scope_as = cmf_log_arg2 }
	scope:actor = { cmf_log_with_scope_arg = { action = ht_log_converted_pops } }
}
