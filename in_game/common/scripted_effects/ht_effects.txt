	scope:recipient = {
		annex_country = {
			country = scope:target
			reason = Diplomatic
			transfer = yes
		}
	}
	scope:recipient = { save_scope_as = cmf_log_arg1 }
	scope:target = { save_scope_as = cmf_log_arg2 }
	scope:actor = { cmf_log_with_scope_args = { action = ht_log_annexed } }
}

# Single-target: recipient = subject to free

	scope:recipient = {

		}
	}
	scope:recipient = { save_scope_as = cmf_log_arg2 }
	scope:actor = { cmf_log_with_scope_arg = { action = ht_log_granted_independence } }
}
