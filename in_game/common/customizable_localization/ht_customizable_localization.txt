# Recipient-slot display name. type = none keeps it null-safe when scope:recipient is null at
# preview time; the ht_target_self branch returns the player's own name because recipient = actor
# in that mode.
ht_get_display_country = {
	type = none
	text = {
		trigger = { has_global_variable = ht_target_self }
		localization_key = ht_self_country_name
	}
	text = {
		trigger = { country_exists = scope:recipient }
		localization_key = ht_recipient_country_name
	}
	text = {
		localization_key = ht_another_country_placeholder
		fallback = yes
	}
}

# Picked-target display name. type = none keeps it null-safe when scope:target is null at preview
# time before the user picks.
ht_get_picked_country = {
	type = none
	text = {
		trigger = { country_exists = scope:target }
		localization_key = ht_picked_country_name
	}
	text = {
		localization_key = ht_another_country_placeholder
		fallback = yes
	}
}
